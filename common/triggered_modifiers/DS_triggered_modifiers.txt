DS_legendary_dynasty_modifier = {
	potential = { 
		government = monarchy
	}
	trigger = {
		calc_true_if = {
			all_countries_including_self = {
				dynasty = ROOT
				OR = {
					is_subject = no
					is_subject_of_type = personal_union
					is_subject_of_type = tributary_state
				}
			}
			amount = 10
		}
		calc_true_if = {
			all_countries_including_self = {
				dynasty = ROOT
				government_rank = 3
			}
			amount = 1
		}
	}
	diplomatic_reputation = 3
	monthly_splendor = 2
	monthly_heir_claim_increase = 0.2
	legitimacy = 0.5
}
