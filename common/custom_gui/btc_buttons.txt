# Beyond the Cape buttons

custom_button = {
	name = feitoria_button
	potential = {
		FROM = { has_government_attribute = can_establish_feitorias  }
		NOT = { has_province_modifier = feitoria }
		NOT = { has_province_modifier = fortified_feitoria }
		owned_by = FROM
		has_port = yes
		OR = {
			AND = {
				continent = africa
				NOT = { region = maghreb_region }
				NOT = { region = egypt_region }
			}
			AND = {
				continent = asia
				NOT = { region = mashriq_region }
				NOT = { region = east_siberia_region }
			}
		}
	}
	trigger = {
		area_for_scope_province = {
			type = all
			OR = {
				NOT = { has_province_modifier = feitoria }	
				NOT = { has_province_modifier = fortified_feitoria }	
			}
		}
		FROM = { 
			if = {
				limit = {
					NOT = { dip_tech = 7 }
				}
				treasury = 20
			}
			else_if = {
				limit = {
					dip_tech = 7
					NOT = { dip_tech = 9 }
				}
				treasury = 40
			}
			else_if = {
				limit = {
					dip_tech = 9
					NOT = { dip_tech = 11 }
				}
				treasury = 60
			}
			else_if = {
				limit = {
					dip_tech = 11
					NOT = { dip_tech = 13 }
				}
				treasury = 80
			}
			else_if = {
				limit = {
					dip_tech = 13
				}
				treasury = 100
			}
		}
	}
	effect = {
		add_province_modifier = {
			name = feitoria
			duration = -1
		}
		FROM = { 
			if = {
				limit = {
					NOT = { dip_tech = 7 }
				}
				add_treasury = -20
			}
			else_if = {
				limit = {
					dip_tech = 7
					NOT = { dip_tech = 9 }
				}
				add_treasury = -40
			}
			else_if = {
				limit = {
					dip_tech = 9
					NOT = { dip_tech = 11 }
				}
				add_treasury = -60
			}
			else_if = {
				limit = {
					dip_tech = 11
					NOT = { dip_tech = 13 }
				}
				add_treasury = -80
			}
			else_if = {
				limit = {
					dip_tech = 13
				}
				add_treasury = -100
			}
		}
	}
	tooltip = feitoria_key
}

custom_button = {
	name = fortified_feitoria_button
	potential = {
		FROM = { has_government_attribute = can_establish_feitorias  }
		has_province_modifier = feitoria
	}
	trigger = {
		FROM = { 
			if = {
				limit = {
					NOT = { dip_tech = 7 }
				}
				treasury = 60
			}
			else_if = {
				limit = {
					dip_tech = 7
					NOT = { dip_tech = 9 }
				}
				treasury = 120
			}
			else_if = {
				limit = {
					dip_tech = 9
					NOT = { dip_tech = 11 }
				}
				treasury = 180
			}
			else_if = {
				limit = {
					dip_tech = 11
					NOT = { dip_tech = 13 }
				}
				treasury = 240
			}
			else_if = {
				limit = {
					dip_tech = 13
				}
				treasury = 300
			}
		}
	}
	effect = {
		remove_province_modifier = feitoria
		add_province_modifier = {
			name = fortified_feitoria
			duration = -1
		}
		FROM = { 
			if = {
				limit = {
					NOT = { dip_tech = 7 }

			}
			else_if = {
				limit = {
					dip_tech = 7

			}
			else_if = {

			}

	}
	tooltip = fortified_feitoria_key
}
