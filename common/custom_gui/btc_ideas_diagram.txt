custom_button = {
    name = btc_ideas_diagram_government
    potential = {
		has_country_flag = btc_showing_ideas_hint
    }
    trigger = {
        hidden_trigger = {
            always = yes
        }
    }
    effect = {
		hidden_effect = {
			if = {
				limit = {
					OR = {
						btc_can_unlock_government_idea_group = { government = aristocratic }
						has_idea_group = aristocracy_ideas
					}
				}
				country_event = {
					id = btc_ideas_diagram.1
				}
			}
			if = {
				limit = {
					OR = {
						btc_can_unlock_government_idea_group = { government = plutocratic }
						has_idea_group = plutocracy_ideas
					}
				}
				country_event = {
					id = btc_ideas_diagram.2
				}
			}
		}
    }
    tooltip = btc_ideas_diagram_empty
    frame = {
        number = 2
        trigger = {
            OR = {
				has_idea_group = aristocracy_ideas
				has_idea_group = plutocracy_ideas
				has_idea_group = horde_gov_ideas
				has_idea_group = theocracy_gov_ideas
				has_idea_group = indigenous_ideas
			}
        }
    }
	frame = {
        number = 1
        trigger = {
			btc_can_unlock_idea_group = { government = yes }
        }
    }
	frame = {
		number = 3
		trigger = {
			always = yes
		}
	}
}
