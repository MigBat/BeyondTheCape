check_settler_pool_effects = {

	# settler pool colonial growth bonus
	set_variable = {
		which = settler_pool_colonial_growth
		which = total_settler_pool_amount
	}
	subtract_variable = {
		which = settler_pool_colonial_growth
		value = 60
	}
	multiply_variable = {
		which = settler_pool_colonial_growth
		value = 1.25
	}
	if = {
		limit = {
			NOT = {
				check_variable = {
					which = settler_pool_colonial_growth
					value = 0
				}
			}
		}
		set_variable = {
			which = settler_pool_colonial_growth
			value = 0
		}
	}
	# settler pool colony cost penalty
	set_variable = {
		which = settler_pool_colony_cost
		which = total_settler_pool_amount
	}
	divide_variable = {
		which = settler_pool_colony_cost
		value = 80
	}
	subtract_variable = {
		which = settler_pool_colony_cost
		value = 1
	}
	multiply_variable = {
		which = settler_pool_colony_cost
		value = -100
	}
	if = {
		limit = {
			NOT = {
				check_variable = {
					which = settler_pool_colony_cost
					value = 0
				}
			}
		}
		set_variable = {
			which = settler_pool_colony_cost
			value = 0
		}
	}
}
