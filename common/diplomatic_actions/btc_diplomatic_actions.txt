# Restrict military access and map sharing

milaccess = {
    condition = {
        tooltip = is_ally_of_enemy_tt
        potential = {
            is_at_war = yes
            FROM = {
                any_ally = { 
                    OR = {
                        is_in_war = { 
                            attackers = ROOT
                            defenders = THIS
                        }
                        is_in_war = { 
                            attackers = THIS
                            defenders = ROOT
                        }
                    }
                }
                NOT = { alliance_with = ROOT }
            }
        }
        allow = { always = no }
    }
	condition = {
		tooltip = does_not_border_tt
		potential = {
			FROM = { 
				NOT = { 
					is_neighbor_of = ROOT
				}
			}
		}
		allow = {
			OR = {
				FROM = { alliance_with = ROOT }
				any_subject_country = {
					is_neighbor_of = FROM
				}
				AND = {
					ROOT = { is_part_of_hre = yes }
					FROM = { is_part_of_hre = yes }
				}
			}
		}
	}
}

sharemap = {
	condition = {
		tooltip = is_colonizer_tt
		potential = {
			FROM = {
				OR = {
					has_idea_group = exploration_ideas
					has_idea_group = expansion_ideas
					num_of_colonists = 1
					tag = POR
					tag = NOR
				}
			}
		}
		allow = { always = no }
	}
	condition = {
		tooltip = low_trust_tt
		potential = {
			FROM = {
				NOT = {
					trust = {
						who = ROOT
						value = 80
					}
				}
			}
		}
		allow = { always = no }
	}
}

callaction = {
	condition = {
		tooltip = ccw_callaction_tt
		potential = {
			has_country_flag = at_war
		}
		allow = { 
			FROM = {
				OR = {
					tag = POR
					tag = ARA
					is_subject_of = POR
					is_subject_of = ARA
					is_subject_of = CAS
				}
			}
		}
	}
	condition = {
		tooltip = plw_callaction_tt
		potential = {
			has_country_flag = plw_at_war
		}
		allow = { 
			FROM = {
				is_subject_of = ROOT
			}
		}
	}
}

requestpeace = {
	condition = {
		tooltip = ccw_requestpeace_tt
		potential = {
			OR = {
				tag = POR
				tag = ARA
			}
			has_country_flag = at_war
		}
		allow = { 
			FROM = { tag = CAS }
		}
	}
}

declarewar = {
	condition = {
		tooltip = GRAND_CRUSADE_NO_OTHER_WAR
		potential = { }
		allow = {
			NOT = {
				is_in_war = {
					attacker_leader = ROOT
					casus_belli = cb_grand_crusade
				}
			}
		}
	}
	effect = {
		if = {
			limit = {
				has_global_flag = grand_crusades_enabled
				has_country_flag = grand_crusader_war_monger_flag
				is_in_war = {
					attacker_leader = ROOT
					casus_belli = cb_grand_crusade
				}
			}
			FROM = {
				save_global_event_target_as = grand_crusade_target
			}
		}
	}
}
