anabaptist_diplomacy_aspect = {
	sprite = "GFX_diplomats_of_anabaptism_aspect"
	cost = 100
	potential = {
		religion = anabaptist
	}
	trigger = {}
	effect = {
		custom_tooltip = diplomats_of_the_reformation_aspect_tt
		hidden_effect = {
			every_known_country = {
				limit = {
					OR = {
						religion = protestant
						religion = reformed
						religion = anglican
					}
				}
				add_opinion = {
					who = ROOT
					modifier = anglican_reformation_diplomacy
				}
				reverse_add_opinion = {
					who = ROOT
					modifier = anglican_reformation_diplomacy
				}
			}
		}
	}
	ai_will_do = {
		factor = 0
	}
}
