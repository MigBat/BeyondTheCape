estate_slavers_disaster = {
	potential = {
		has_estate = estate_slavers
		estate_influence = {
			estate = estate_slavers
			influence = 80
		}
		NOT = { has_disaster = tur_disaster_ottoman_power_struggle }
	}


	can_start = {
		has_any_disaster = no
		estate_influence = {
			estate = estate_slavers
			influence = 100
		}
		is_at_war = no
	}
	
	can_stop = {
		OR = {
			has_any_disaster = yes
			NOT = {
				estate_influence = {
					estate = estate_slavers
					influence = 100
				}		
			}
		}
	}
	
	down_progress = {
		factor = 0
		modifier = {
			factor = 1
			NOT = {
				estate_influence = {
					estate = estate_slavers
					influence = 100
				}
			}
		}
	}
	
	progress = {
		modifier = {
			factor = 0.5
			NOT = {
				estate_loyalty = {
					estate = estate_slavers
					loyalty = 60
				}
			}
			is_at_war = no
		}
		modifier = {
			factor = 0.5
			NOT = {
				estate_loyalty = {
					estate = estate_slavers
					loyalty = 50
				}
			}
			is_at_war = no
		}
		modifier = {
			factor = 0.5
			NOT = {
				estate_loyalty = {
					estate = estate_slavers
					loyalty = 40
				}
			}
			is_at_war = no
		}
		modifier = {
			factor = 1
			NOT = {
				estate_loyalty = {
					estate = estate_slavers
					loyalty = 30
				}
			}
			is_at_war = no
		}
		modifier = {
			factor = 1
			NOT = {
				estate_loyalty = {
					estate = estate_slavers
					loyalty = 20
				}
			}
			is_at_war = no
		}
	}
	
	can_end = {
		custom_trigger_tooltip = {
			tooltip = EST_CRUSHED_SLAVERS
			OR = {
				NOT = { has_country_flag = slavers_estate_in_power }
				has_country_flag = noble_estate_in_power
				has_country_flag = church_estate_in_power
			}						
		}
	}
	
	modifier = {	
		reduced_liberty_desire_on_other_continent = -100
		global_trade_goods_size = -0.5
		global_unrest = 2
		global_ship_trade_power = -0.5
	}

	on_start = estate_disasters.30
	on_end = estate_disasters.31
	
	on_monthly = {
	}
}
