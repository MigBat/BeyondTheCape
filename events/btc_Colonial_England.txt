#Australia
country_event = {
	id = btc_colonial_england.1
	title = "btc_colonial_england.1.t"
	desc = "btc_colonial_england.1.d"
	picture = COLONIZATION_eventPicture
	
	is_triggered_only = yes
	hidden = yes
	
	trigger = {
		capital_scope = {
			colonial_region = colonial_australia
			is_core = ROOT
		}
		NOT = { exists = E00 }
		OR = {
			is_colonial_nation_of = ENG
			is_colonial_nation_of = GBR
			is_colonial_nation_of = AVE
		}
	}
	option = {
		name = "btc_colonial_england.a"
		swap_free_idea_group = yes
		change_tag = E00
		swap_non_generic_missions = yes
		change_religion = overlord
		btc_update_colonial_australia_colour = yes
	
	}
}
