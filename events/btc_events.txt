	}
}

# Irmandinos Revolt #

country_event = {
	id = btc_events.3
	title = btc_events.3_t
	desc = btc_events.3_d
	picture = BORDER_TENSION_eventPicture

	fire_only_once = yes

	trigger = {
		tag = CAS
	}

	mean_time_to_happen = {
		months = 60
	}

	option = {	# Help the local nobility fight the peasantry
		name = btc_events.3a

		add_years_of_income = -0.5
		add_manpower = -10
		add_country_modifier = {
			name = galician_revolts	# +5 global unrest
			duration = 1825
		}
		
		every_owned_province = {
			limit = {
				area = galicia_area
			}
			add_devastation = 50
		}
	}

	option = {	# Let the nobles fight the peasants
		name = btc_events.3b

		every_owned_province = {
			limit = { has_province_modifier = strong_nobility }
			remove_province_modifier = strong_nobility
		}
		create_vassal = GAL
		GAL = { 
			add_liberty_desire = 100
			add_manpower = 10
			country_event = {
				id = btc_events.4
				days = 30
			}
		}
	}
}
