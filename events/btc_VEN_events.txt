
  #Foscari's deposition by the council of ten 

  namespace = btc_ven
normal_or_historical_nations = yes

country_event = {	
	id = btc_ven.2
	title = btc_ven.2_t
	desc = btc_ven.2_d
	picture = SPY_eventPicture

	fire_only_once = yes
	
	trigger = {
		tag = VEN
		is_year = 1450
		NOT = { is_year = 1500 }
		has_government_attribute = is_merchant_republic
		NOT = {
			OR = {
				has_country_modifier = btc_ven_ten_emergency_government 
				has_country_modifier = btc_ven_ten_take_control_of_the_chancery 
				has_country_modifier = btc_ven_ten_council_eliminated
				has_country_modifier = btc_ven_ten_council_restored_to_previous_functions
				has_country_modifier = btc_ven_tens_improved_control_of_the_chancery
			}
	    }
	}
	
	mean_time_to_happen = {
		months = 100
		modifier = {
			factor = 0.5
			NOT = { stability = 0 }
		}
	}	
	option = {	#Depose the Doge
		name = btc_ven.2_a
		ai_chance = { 
			factor = 1		
		}
		add_republican_tradition = -15
		add_country_modifier = {
			name = btc_ven_ten_emergency_government
			duration = 1825
		}
		set_global_flag = btc_ven_doge_deposed_by_ten
		custom_tooltip = btc_ven_doge_deposed_by_ten_tt
		kill_ruler = yes
	}
	option = {	#Oppose the ten council
		name = btc_ven.2_b
		ai_chance = { 
			factor = 0	
		}
		add_stability = -1
		add_dip_power = -50
		add_adm_power = -50
		}
	}
