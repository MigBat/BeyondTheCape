	is_triggered_only = yes

	option = {
		name = dutch_domination.9.a
		custom_tooltip = dutch_domination.9.a.tt
		custom_tooltip = voc_province_warning_tt
		hidden_effect = {
			every_owned_province = {
				limit = {
					OR = {
					region = moluccas_region
					region = indonesia_region
					}
				}
				add_core = VOC
				if = {
					limit = { is_core = ROOT }
					remove_core = ROOT
				}
			}
			release = VOC
			create_subject = {
				subject_type = commercial_enterprise
				who = VOC
			}
			VOC = {
				change_government = republic
				add_government_reform = trade_company_government
				change_religion = ROOT
				change_primary_culture = ROOT
				define_ruler = {
					name = "Heren XVIII"
					adm = 3
					dip = 3
					mil = 3
					#regent = yes
				}
				add_adm_power = 100
				add_dip_power = 100
				add_mil_power = 100
				add_treasury = 200
				set_prestige_to_zero = yes
				add_republican_tradition = 100
			}
			country_event = { id = dutch_domination.901 days = 2 }
		}
		moluccas_region = {
			limit = {
				NOT = { is_core = VOC }
				NOT = { is_permanent_claim = VOC }
			}
			add_permanent_claim = VOC
		}
		indonesia_region = {
			limit = {
				NOT = { is_core = VOC }
				NOT = { is_permanent_claim = VOC }
			}
			add_permanent_claim = VOC
		}
		custom_tooltip = charter_voc_tt
		ai_chance = {
			factor = 1
		}
	}

	option = {
		name = dutch_domination.9.b

			limit = {
				NOT = { is_core = ROOT }
				NOT = { is_permanent_claim = ROOT }
			}
			add_permanent_claim = ROOT
		}
		indonesia_region = {
			limit = {
				NOT = { is_core = ROOT }
				NOT = { is_permanent_claim = ROOT }
			}
			add_permanent_claim = ROOT
		}
		ai_chance = {
			factor = 0
		}
	}
	
	#Play as the East India Company
	option = {
		name = dutch_domination.9.c
		trigger = {
			ai = no
