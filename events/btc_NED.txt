# [Root.Province.GetName] to Commercial Enterprise
province_event = {
	id = btc_ned.2
	title = btc_ned.2_t
	desc = btc_ned.2_d
	picture = 16_Trade_Company_eventPicture

	is_triggered_only = yes

	option = {	# Give it to our commercial enterprise
		name = btc_ned.2a

		ai_chance = {
			factor = 1
		}

		if = {
			limit = {
				VOC = {
					is_subject_of = root
				}
			}
			cede_province = VOC			
		}
		else_if = {
			limit = {
				EIC = {
					is_subject_of = root
				}
			}
			cede_province = EIC
		}
	}

	option = {	# Administer it directly
		name = btc_ned.2b

		ai_chance = {
			factor = 0
		}
	}
}

# Adolf's Revolt
country_event = {
	id = btc_ned.3
	title = btc_ned.3_t
	desc = btc_ned.3_d
	picture = CONSPIRACY_eventPicture

	trigger = {
		tag = BUR
		GEL = {
			exists = yes
			has_ruler = "Arnold"
			has_heir = "Adolf"
			heir_age = 15
			is_subject = no
		}
	}

	fire_only_once = yes

	mean_time_to_happen = {
		months = 60
	}

	option = {	# Support Adolf
		name = btc_ned.3a

		ai_chance = {
			factor = 1
		}

		add_treasury = -50

		GEL = {
			kill_ruler = yes
			country_event = {
				id = btc_ned.4
				days = 30
			}
		}
	}

	option = {	# He is not trustworthy
		name = btc_ned.3b

		ai_chance = {
			factor = 0
		}

		add_prestige = -5
	}
}
