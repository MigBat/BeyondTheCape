#John Cabot
country_event = {
	id = btc_england.1
	title = btc_england.1_t
	desc = btc_england.1_d
	picture = SHIP_SAILING_eventPicture
	
	fire_only_once = yes
	
	trigger = {
		OR = {
			tag = ENG
			AND = {
				was_tag = ENG
				NOT = { exists = ENG }
				AND = {
					tag = GBR
					tag = AVE
					}
				}
			}
		NOT = { has_country_modifier = quest_for_the_new_world }
		any_country = {
			had_country_flag = {
				flag = had_columbus
				days = 2920
			}
		}
	}

	mean_time_to_happen = {
		months = 120
		modifier = {
			factor = 0.1
			has_idea_group = exploration_ideas
		}
	}
	
	option = {		# Hire Cabot
		name = btc_england.1a
		
		ai_chance = { factor = 1 }

		add_years_of_income = -0.25
		set_global_flag = columbus_happened
		set_country_flag = had_columbus
		hidden_effect = {
			ROOT = {
				country_event = {
					id = btc_england.2
					days = 90
				}
			}
		}
	}
	
	option = {		# Reject his proposal
		name = btc_england.1b
		
		ai_chance = { factor = 0 }
		add_adm_power = 50
		add_years_of_income = 0.5

	}
}
